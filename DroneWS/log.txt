2025/12/24 V2b
	Mavros安裝
		安装 Mavros 驱动
			echo "source ~/ros2_ws/install/setup.bash" >> ~/.bashrc
			source ~/.bashrc
		
		检查 Mavros 包是否存在
			ros2 pkg list | grep mavros
			输出含mavros、mavros_extras等包名则安装成功
			
		查看 Mavros 状态
			ros2 topic echo /mavros/state

		# 查看MAVROS相关包
			ros2 pkg list | grep mavros
			只要输出mavros、mavros_msgs、mavros_extras，即说明安装成功
			
		使用 MAVROS（启动节点示例）
			1. 串口权限配置（解决权限不足问题）
				# 添加当前用户到dialout组, 重启终端或重新登录生效
				sudo usermod -aG dialout $USER

			2. 启动 MAVROS 节点 根据飞控类型选择对应命令：
				# 1. PX4飞控（USB串口：/dev/ttyUSB0，波特率57600）
				ros2 launch mavros px4.launch.py fcu_url:=/dev/ttyUSB0:57600

				# 2. ArduPilot飞控（树莓派硬件串口：/dev/ttyAMA0，波特率115200）
				ros2 launch mavros apm.launch.py fcu_url:=/dev/ttyAMA0:115200

				# 3. 仿真环境（SITL，无需硬件串口）
				ros2 launch mavros px4.launch.py fcu_url:=udp://:14540@127.0.0.1:14557
			成功启动标志：终端无报错，显示mavros node started等信息。
